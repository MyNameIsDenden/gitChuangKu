<body>
    <table>
        <thead>
            <tr>
                <th>姓名</th>
                <th>科目</th>
                <th>分数</th>
                <th>操作</th>
            </tr>
        </thead>
        <tbody>
            
        </tbody>
    </table>
    <script>
        var datas=[
            {
                name:'aaa',
                subject:'zzz',
                score:'123'
            },
            {
                name:'bbb',
                subject:'zzz',
                score:'13'
            },
            {
                name:'ccc',
                subject:'zzz',
                score:'23'
            },
            {
                name:'ddd',
                subject:'zzz',
                score:'231'
            }
        ];
        var h=document.querySelector('tbody');
        for(var i=0;i<datas.length;i++){
            var b=document.createElement('tr');
            h.appendChild(b);
            for(var k in datas[i]){
                var z=document.createElement('td');
                z.innerHTML=datas[i][k];
                b.appendChild(z);
            }
            var z=document.createElement('td');
            z.innerHTML='<a href="javascript:;">删除</a>';
            b.appendChild(z);
        }
        var as=document.querySelectorAll('a');
        for(var i=0;i<as.length;i++){
            as[i].onclick=function(){
                h.removeChild(this.parentNode.parentNode);
            }
        }
    </script>
</body>
