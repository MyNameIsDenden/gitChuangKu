**发送短信案例**

```
<div>
        <button>发送</button>
    </div>
    <script>
        var btn=document.querySelector('button');
        btn.addEventListener('click',function(){
            var time=3;
            var timer=setInterval(function(){
                if(time==0){
                    btn.disabled=false;
                    btn.innerHTML='发送';
                    clearInterval(timer);
                    time=3;
                }else{
                    btn.disabled=true;
                    btn.innerHTML='还剩下'+time+'重新发送';
                    time--;
                }
            },1000);        
        });
    </script>
```
