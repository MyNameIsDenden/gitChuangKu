

1. 创建对象

```
var obj={
	uname='dwada',//用逗号隔开，最后一位不用
	sayHi:function(){       ‘:’加函数
	console.log('HI');
	}
}
//obj.uname;或obj['age'];
//obj.sayHi();不要忘记括号//函数
```

```js
var obj=new Object();//创建一个空对象
        obj.uname='dwa';
        obj.sex='nan';
        obj.sayHi=function(){
            console.log('hi~');
        }
```

```js
function Star(uname,age,sex){
            this.name=uname;
            this.age=age;
            this.sex=sex;
    		this.sing=function(sang){
				console.log(sang);
            }
        }
        var ldh=new Star('刘德华',18,'男');
        console.log(typeof ldh);//Object
        console.log(ldh.name);
		console.log(ldh['sex']);
		ldh.sing('冰雨');
```
